GHISA V1 — Schema Diagram

SwiftData on-device data model. All IDs are UUIDs. Hover to highlight relationships.

erDiagram

    %% ── 1. User Profile ──────────────────────────
    users {
        UUID id PK
        String name
        Int age
        String gender
        Float height_cm
        Float weight_kg
        String unit_system
        DateTime created_at
        DateTime updated_at
    }

    %% ── 2. Training Module ───────────────────────
    exercises {
        UUID id PK
        UUID user_id FK
        String name
        StringArray muscle_groups
        String movement_type
        Boolean is_archived
        DateTime created_at
        DateTime updated_at
    }

    exercise_field_definitions {
        UUID id PK
        UUID exercise_id FK
        String name
        String field_type
        String unit
        StringArray select_options
        String system_key
        Int sort_order
        Boolean is_active
        Boolean is_default
        DateTime created_at
    }

    workouts {
        UUID id PK
        UUID user_id FK
        String status
        Date date
        DateTime started_at
        DateTime ended_at
        Int duration_minutes
        Text notes
        String location
        DateTime created_at
        DateTime updated_at
    }

    workout_templates {
        UUID id PK
        UUID user_id FK
        String name
        Text notes
        DateTime created_at
        DateTime updated_at
    }

    workout_template_exercises {
        UUID id PK
        UUID template_id FK
        UUID exercise_id FK
        Int sort_order
        Int superset_group
        Int target_sets
        Text notes
    }

    workout_template_field_targets {
        UUID id PK
        UUID template_exercise_id FK
        UUID field_definition_id FK
        Float target_value_number
        String target_value_text
    }

    workout_exercises {
        UUID id PK
        UUID workout_id FK
        UUID exercise_id FK
        Int sort_order
        Int superset_group
        Text notes
        DateTime created_at
    }

    workout_sets {
        UUID id PK
        UUID workout_exercise_id FK
        Int set_number
        Text notes
        DateTime created_at
    }

    workout_set_values {
        UUID id PK
        UUID workout_set_id FK
        UUID field_definition_id FK
        Float value_number
        String value_text
        Boolean value_toggle
    }

    flags {
        UUID id PK
        UUID user_id FK
        String name
        String color
        String icon
        String scope
        DateTime created_at
    }

    flag_assignments {
        UUID id PK
        UUID flag_id FK
        UUID workout_id FK
        UUID workout_exercise_id FK
        UUID workout_set_id FK
        DateTime created_at
    }

    food_item_nutrients {
        UUID id PK

    recipes {

        String name

        UUID id PK

    %% ── Relationships ────────────────────────────

    %% User owns everything
    users ||--o{ exercises : "has"
    users ||--o{ workouts : "has"
    users ||--o{ workout_templates : "has"
    users ||--o{ flags : "has"
    users ||--o{ nutrient_definitions : "defines"
    users ||--o{ food_items : "creates"
    users ||--o{ recipes : "has"
    users ||--o{ meal_categories : "has"
    users ||--o{ meal_entries : "has"
    users ||--o{ meal_templates : "has"
    users ||--o{ nutrition_targets : "has"
    users ||--o{ daily_log_field_definitions : "defines"
    users ||--o{ daily_logs : "has"
    users ||--o{ correlation_results : "has"

    %% Training
    exercises ||--o{ exercise_field_definitions : "defines"
    exercises ||--o{ workout_exercises : "used in"
    exercises ||--o{ workout_template_exercises : "used in"

    workouts ||--o{ workout_exercises : "contains"
    workout_exercises ||--o{ workout_sets : "contains"
    workout_sets ||--o{ workout_set_values : "has"
    exercise_field_definitions ||--o{ workout_set_values : "defines"

    workout_templates ||--o{ workout_template_exercises : "contains"
    workout_template_exercises ||--o{ workout_template_field_targets : "has"
    exercise_field_definitions ||--o{ workout_template_field_targets : "defines"

    flags ||--o{ flag_assignments : "applied via"
    workouts ||--o{ flag_assignments : "tagged"
    workout_exercises ||--o{ flag_assignments : "tagged"
    workout_sets ||--o{ flag_assignments : "tagged"

    %% Nutrition
    nutrient_definitions ||--o{ food_item_nutrients : "values"
    nutrient_definitions ||--o{ nutrition_targets : "target for"
    food_items ||--o{ food_item_nutrients : "has"

    recipes ||--o{ recipe_ingredients : "contains"
    food_items ||--o{ recipe_ingredients : "used in"

    meal_categories ||--o{ meal_entries : "groups"
    food_items ||--o{ meal_entries : "logged as"
    recipes ||--o{ meal_entries : "logged as"

    meal_categories ||--o{ meal_templates : "defaults to"
    meal_templates ||--o{ meal_template_items : "contains"
    food_items ||--o{ meal_template_items : "used in"
    recipes ||--o{ meal_template_items : "used in"

    %% Lifestyle
    daily_logs ||--o{ daily_log_values : "has"
    daily_log_field_definitions ||--o{ daily_log_values : "defines"
